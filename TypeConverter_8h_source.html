184 template <typename T, Type... Types>
185 struct TypeConverterNative {
186  using ValueType = T;
187  using NativeType = T;
188  using ValidTypes = TypeList<Types...>;
189 
190  static_assert(ValidTypes::size() >= 1);
191 
192  static void fromNative(const T& src, T& dst) {
193  if constexpr (std::is_fundamental_v<T>) {
194  dst = src; // shallow copy
195  } else {
196  // just take first type -> underlying memory layout of all types should be the same
197  constexpr auto typeGuess = ValidTypes::toArray().at(0);
198  // clear first
199  UA_clear(&dst, getUaDataType<typeGuess>());
200  // deep copy
201  const auto status = UA_copy(&src, &dst, getUaDataType<typeGuess>());
202  throwOnBadStatus(status);
203  }
204  }
205 
206  static void toNative(const T& src, T& dst) {
207  fromNative(src, dst);
208  }
209 };
