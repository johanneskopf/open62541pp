201template <typename T, TypeIndex... typeIndexes>
202struct TypeConverterNative {
203 using ValueType = T;
204 using NativeType = T;
205 using ValidTypes = TypeIndexList<typeIndexes...>;
206
207 static_assert(ValidTypes::size() >= 1);
208
209 static void fromNative(const T& src, T& dst) {
210 if constexpr (std::is_fundamental_v<T>) {
211 dst = src;
212 } else {
213 // just take first type -> underlying memory layout of all types should be the same
214 constexpr auto typeIndexGuess = ValidTypes::toArray().at(0);
215 // clear first
216 UA_clear(&dst, getUaDataType<typeIndexGuess>());
217 // deep copy
218 const auto status = UA_copy(&src, &dst, getUaDataType<typeIndexGuess>());
219 throwOnBadStatus(status);
220 }
221 }
222
223 static void toNative(const T& src, T& dst) {
224 fromNative(src, dst);
225 }
226};
