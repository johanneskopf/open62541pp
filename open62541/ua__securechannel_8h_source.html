77 struct UA_SecureChannel {
78  UA_SecureChannelState state;
79  UA_SecureChannelRenewState renewState;
80  UA_MessageSecurityMode securityMode;
81  UA_ConnectionConfig config;
82 
83  /* Rules for revolving the token with a renew OPN request: The client is
84  * allowed to accept messages with the old token until the OPN response has
85  * arrived. The server accepts the old token until one message secured with
86  * the new token has arrived.
87  *
88  * We recognize whether nextSecurityToken contains a valid next token if the
89  * ChannelId is not 0. */
90  UA_ChannelSecurityToken securityToken; /* Also contains the channelId */
91  UA_ChannelSecurityToken altSecurityToken; /* Alternative token for the rollover.
92  * See the renewState. */
93 
94  /* The endpoint and context of the channel */
95  const UA_SecurityPolicy *securityPolicy;
96  void *channelContext; /* For interaction with the security policy */
97  UA_Connection *connection;
98 
99  /* Asymmetric encryption info */
100  UA_ByteString remoteCertificate;
101  UA_Byte remoteCertificateThumbprint[20]; /* The thumbprint of the remote certificate */
102 
103  /* Symmetric encryption nonces. These are used to generate the key material
104  * and must not be reused once the keys are in place.
105  *
106  * Nonces are also used during the CreateSession / ActivateSession
107  * handshake. These are not handled here, as the Session handling can
108  * overlap with a RenewSecureChannel. */
109  UA_ByteString remoteNonce;
110  UA_ByteString localNonce;
111 
112  UA_UInt32 receiveSequenceNumber;
113  UA_UInt32 sendSequenceNumber;
114 
115  /* Sessions that are bound to the SecureChannel */
116  SLIST_HEAD(, UA_SessionHeader) sessions;
117 
118  /* If a buffer is received, first all chunks are put into the completeChunks
119  * queue. Then they are processed in order. This ensures that processing
120  * buffers is reentrant with the correct processing order. (This has lead to
121  * problems in the client in the past.) */
122  UA_ChunkQueue completeChunks; /* Received full chunks that have not been
123  * decrypted so far */
124  UA_ChunkQueue decryptedChunks; /* Received chunks that were decrypted but
125  * not processed */
126  size_t decryptedChunksCount;
127  size_t decryptedChunksLength;
128  UA_ByteString incompleteChunk; /* A half-received chunk (TCP is a
129  * streaming protocol) is stored here */
130 
131  UA_CertificateVerification *certificateVerification;
132  UA_StatusCode (*processOPNHeader)(void *application, UA_SecureChannel *channel,
133  const UA_AsymmetricAlgorithmSecurityHeader *asymHeader);
134 };
