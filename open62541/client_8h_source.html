199 static UA_StatusCode
200 UA_Client_connectUsername(UA_Client *client, const char *endpointUrl,
201  const char *username, const char *password) {
202  UA_UserNameIdentityToken* identityToken = UA_UserNameIdentityToken_new();
203  if(!identityToken)
204  return UA_STATUSCODE_BADOUTOFMEMORY;
205  identityToken->userName = UA_STRING_ALLOC(username);
206  identityToken->password = UA_STRING_ALLOC(password);
207  UA_ClientConfig *cc = UA_Client_getConfig(client);
208  UA_ExtensionObject_clear(&cc->userIdentityToken);
209  cc->userIdentityToken.encoding = UA_EXTENSIONOBJECT_DECODED;
210  cc->userIdentityToken.content.decoded.type = &UA_TYPES[UA_TYPES_USERNAMEIDENTITYTOKEN];
211  cc->userIdentityToken.content.decoded.data = identityToken;
212  return UA_Client_connect(client, endpointUrl);
213 }
